# User Database
class UserDatabase:
    def __init__(self):
        # Initialize an empty dictionary to store user credentials
        self.users = {}

    def add_user(self, email, password):
        # Store user credentials in the dictionary
        self.users[email] = password

    def get_user(self, email):
        # Return the password associated with the given email
        return self.users.get(email)


# Auth Service
class AuthService:
    def __init__(self, user_database):
        # Initialize the user database
        self.user_database = user_database

    def login(self, email, password):
        # Check if the email and password are valid
        if email in self.user_database.users and self.user_database.users[email] == password:
            # Generate a token if the credentials are valid
            token = self.generate_token(email)
            return token
        else:
            # Return None if the credentials are invalid
            return None

    def generate_token(self, email):
        # Generate a unique token for the user
        import secrets
        import string
        token = ''.join(secrets.choice(string.ascii_letters + string.digits) for _ in range(16))
        return token


# Main Function
def main():
    # Create a user database and auth service
    user_database = UserDatabase()
    auth_service = AuthService(user_database)

    # Add users to the database
    user_database.add_user("user1@example.com", "password1")
    user_database.add_user("user2@example.com", "password2")

    # Test the login function
    token = auth_service.login("user1@example.com", "password1")
    if token:
        print(f"Login successful. Token: {token}")
    else:
        print("Invalid email or password")


if __name__ == "__main__":
    main()
